create table tasks(
id int AUTO_INCREMENT,
description varchar(255),
complete boolean,
primary key(id)
);

    uid varchar(10) not null,
	userName varchar(255),
	password varchar(255),

	phNo varchar(255),
	userRole int not null,
	primary key(uid)
);

create table images(
	iid int not null,
	fileName varchar(255),
	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
